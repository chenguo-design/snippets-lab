git基础命令：

    git remote add <name> <url>         建立与远程仓库连接
    git remote rm <name>                取消与远程仓库建立连接

    git log --pretty=oneline
    git reset --hard 81c9aa
    git push -f                         强制回退版本

    git revert -n 81c9aa
    git commit -m "xxx"
    git push                            撤销某个版本
